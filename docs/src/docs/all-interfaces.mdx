### ITaskResponse

```ts
interface ITaskResponse<T extends ITaskSolution>
    extends IErrorResponse,
        Pick<IResponse, "status"> {
    solution?: T
}
```

### ITaskRequest

```ts
interface ITaskRequest<T extends ITask> extends Pick<IRequest, "clientKey"> {
    task: T
}
```

## type

### WithRequired

```ts
type WithRequired<T, K extends keyof T> = T & { [P in K]-?: T[P] }
```

### ProxyType

```ts
type ProxyType = "http" | "https" | "socks4" | "socks5"
```

### PartialBy

```ts
type PartialBy<T, K extends keyof T> = Omit<T, K> & Partial<Pick<T, K>>
```
